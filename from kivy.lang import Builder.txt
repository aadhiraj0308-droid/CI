name: CI - Tests & Build

on:
  push:
    branches: [ main, master ]
  pull_request:
    branches: [ main, master ]

jobs:
  test-and-build:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:15
        env:
          POSTGRES_USER: ci_user
          POSTGRES_PASSWORD: ci_pass
          POSTGRES_DB: ci_db
        # no host port mapping required for Actions service; runner can reach it at localhost:5432
        options: >-
          --health-cmd "pg_isready -U ci_user -d ci_db"
          --health-interval 10s --health-timeout 5s --health-retries 5

    env:
      # used by backend to run tests against the service
      DATABASE_URL: postgres://ci_user:ci_pass@localhost:5432/ci_db
      # set any other ENV variables your tests require here, e.g. SECRET_KEY: test-secret

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Install pg_isready (postgres client)
        run: |
          # ensure pg_isready is available on the runner
          sudo apt-get update -y
          sudo apt-get install -y postgresql-client

      - name: Wait for Postgres to be healthy
        run: |
          echo "Waiting for Postgres to be healthy..."
          for i in {1..30}; do
            if pg_isready -h localhost -p 5432 -U ci_user -d ci_db >/dev/null 2>&1 ; then
              echo "Postgres is ready!"
              break
            fi
            echo "Postgres not ready yet... ($i)"
            sleep 2
          done

      - name: Set up Python 3.11
        uses: actions/setup-python@v4
        with:
          python-version: "3.11"

      - name: Install backend deps
        working-directory: ./backend
        run: |
          python -m pip install --upgrade pip
          if [ -f "requirements-dev.txt" ]; then
            pip install -r requirements-dev.txt
          elif [ -f "requirements.txt" ]; then
            pip install -r requirements.txt
          else
            echo "No requirements file found in ./backend — skipping pip install."
          fi

      - name: Run DB migrations (optional)
        if: ${{ always() }}
        working-directory: ./backend
        run: |
          # uncomment / update the command below if you use alembic / django etc.
          # alembic upgrade head
          # or for Django: python manage.py migrate --noinput
          echo "Run migrations here if you have migrations configured."

      - name: Run backend tests (pytest)
        working-directory: ./backend
        run: |
          pytest -q --maxfail=1

      - name: Set up Node 20
        uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: Install & build frontend
        working-directory: ./frontend
        run: |
          if [ -f package.json ]; then
            npm ci
            npm run lint || true
            npm run build
          else
            echo "No frontend/package.json found in ./frontend — skipping frontend build."
          fi

      - name: (Optional) Run Playwright E2E (if config exists)
        working-directory: ./frontend
        run: |
          # This step will only run Playwright if a config file exists.
          if [ -f "playwright.config.ts" ] || [ -f "playwright.config.js" ] || grep -q "playwright" package.json 2>/dev/null; then
            npm ci
            npx playwright install --with-deps
            npx playwright test --reporter=list
          else
            echo "Playwright not detected — skipping E2E."
          fi
